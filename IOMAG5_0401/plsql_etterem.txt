SELECT x.ekod,
       x.nev,
       x.varos,
       x.utca,
       x.csillag
from vendeglatas_xml_2 v,
    XMLTABLE('/vendeglatas/etterem'
           PASSING v.adat
           COLUMNS
                ekod    VARCHAR2(10)    PATH    '@eKod',
                nev     VARCHAR2(200)   PATH    'nev',
                varos   VARCHAR2(100)   PATH    'cim/varos',
                utca    VARCHAR2(200)   PATH    'cim/utca',
                hazszam VARCHAR2(10)    PATH    'cim/hazszam',
                csillag NUMBER          PATH    'csillag'
    ) x;

// Task 1/b

SELECT
    x.ekod,
    x.nev,
    x.varos,
    x.csillag
FROM
    vendeglatas_xml_2 v,
    XMLTABLE('/vendeglatas/etterem'
        PASSING v.adat
        COLUMNS
            ekod    VARCHAR2(10)  PATH '@ekod',
            nev     VARCHAR2(200) PATH 'nev',
            varos   VARCHAR2(200) PATH 'cim/varos',
            csillag NUMBER        PATH 'csillag'
    ) x
WHERE x.csillag = 5;

// Task 1/c

SELECT
    sz.szkod,
    sz.nev      AS szakacs_neve,
    sz.reszleg,
    sz.eletkor,
    e.nev       AS etterem_neve
FROM
    vendeglatas_xml_2 v,
    XMLTABLE('/vendeglatas/szakacs'
        PASSING v.adat
        COLUMNS
            szkod   VARCHAR2(10)  PATH '@szkod',
            nev     VARCHAR2(200) PATH 'nev',
            reszleg VARCHAR2(200) PATH 'reszleg',
            eletkor NUMBER        PATH 'eletkor',
            e_sz    VARCHAR2(10)  PATH '@e_sz'
    ) sz
JOIN
    XMLTABLE('/vendeglatas/etterem'
        PASSING v.adat
        COLUMNS
            ekod VARCHAR2(10)  PATH '@ekod',
            nev  VARCHAR2(200) PATH 'nev'
    ) e
ON sz.e_sz = e.ekod;

// Task 2/a

CREATE OR REPLACE TYPE cim_t AS OBJECT (
    varos   VARCHAR2(100),
    utca    VARCHAR2(200),
    hazszam VARCHAR2(20)
);

// Task 2/b

CREATE OR REPLACE TYPE etterem_t AS OBJECT (
    ekod    VARCHAR2(10),
    nev     VARCHAR2(200),
    cim     cim_t,           -- nested object type
    csillag NUMBER,

    MEMBER FUNCTION leiras RETURN VARCHAR2
);

//Task 2/c

CREATE OR REPLACE TYPE BODY etterem_t AS
    MEMBER FUNCTION leiras RETURN VARCHAR2 IS
    BEGIN
        RETURN SELF.nev || ' (' || SELF.csillag || ' csillag) – ' || SELF.cim.varos;
    END;
END;

//Task 3/a

CREATE TABLE ettermek OF etterem_t (
       PRIMARY KEY (ekod)
);

//Task 3/b

INSERT INTO ettermek
SELECT etterem_t(
        x.ekod,
        x.nev,
        cim_t(x.varos,x.utca,x.hazszam),
        x.csillag
    ) from vendeglatas_xml_2 v,
    XMLTABLE('/vendeglatas/etterem'
           PASSING v.adat
           COLUMNS
                ekod    VARCHAR2(10)    PATH    '@ekod',
                nev     VARCHAR2(200)   PATH    'nev',
                varos   VARCHAR2(100)   PATH    'cim/varos',
                utca    VARCHAR2(200)   PATH    'cim/utca',
                hazszam VARCHAR2(10)    PATH    'cim/hazszam',
                csillag NUMBER          PATH    'csillag'
    ) x;

SELECT
    e.cim.varos        AS varos,
    COUNT(*)           AS db,
    AVG(e.csillag)     AS atlag_csillag
FROM ettermek e
GROUP BY e.cim.varos
ORDER BY atlag_csillag DESC;
