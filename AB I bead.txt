create table TAGOK(
	ID int primary key,
	nev varchar2(30),
	ir_szam int(4),
	utca varchar2(20),
	hazszam char(5)
);

create table SZAMITOGEP(
	sz_szam int primary key,
	op_rendszer varchar1(15),
	tarhely int(4)
);

create table NYELVEK(
	NyID int primary key,
	nyelv varchar2(10)
);

create table FILMEK(
	FID int primary key,
	cim varchar2(20),
	kiado varchar2(20),
	megj_ev date,
	hossz int(3)
);

create table KONYVEK(
	BID int primary key,
	szerzo varchar2(20),
	cim varchar2(20),
	kiado varchar2(20),
	kiad_ev date
);

create table FELSZERELT(
	billentyuzet varchar2(15),
	eger varchar2(10),
	sz_szam int primary key,
	foreign key(sz_szam) references SZAMITOGEP(sz_szam) 
	
);

 create table DALOK(
         DID int primary key,
         cim varchar2(20) NOT NULL,
         hossz int NOT NULL,
         megj_ev int,
         nyelv varchar2(10)
 );

create table ELOADOK(
	DID int primary key,
	eloado_1 varchar2 (20) NOT NULL,
	eloado_2 varchar2 (20),
	foreign key (DID) references DALOK(DID)
);

create table ALBUMOK(
	AID int primary key,
	kiado varchar2(20),
	kiad_ev int,
	cim varhcar2(20)
);

create table TARTALMAZ(
	AID int,
	DID int,
	primary key(AID,DID),
	foreign key(AID) references ALBUMOK(AID),
	foreign key(DID) references DALOK(DID)
);

create table HASZNALAT(
	HID int primary key,
	haszn_ido date NOT NULL,
	datum date NOT NULL,
	sz_szam int,
	ID int,
	foreign key(sz_szam) references SZAMITOGEP(sz_szam),
	foreign key(ID) references TAGOK(ID)
);

create table BKOLCSON(
	BK int primary key,
	kezd_datum date NOT NULL,
	vissza_datum date,
	ID int,
	BID int,
	foreign key(BID) references KONYVEK(BID),
	foreign key(ID) references TAGOK(ID)
);

create table IGENYLES(
	ID int,
	BID int,
	igeny_datum date NOT NULL,
	szalllit_datum date,
	beszerz_ar int,
	primary key(BID,ID),

	foreign key(ID) references TAGOK(ID),
	foreign key(BID) references KONYVEK(BID)
);

create table FKOLCSON(
	FK int primary key,
	kezd_datum date NOT NULL,
	vissza_datum date,
	ID int,
	FID int,
	foreign key(ID) references TAGOK(ID),
	foreign key(FID) references FILMEK(FID)
); 

create table AKOLCSON(
	AK int primary key,
	kezd_datum date NUT NULL,
	vissza_datum date,
	ID int,
	AID int,
	foreign key(ID) references TAGOK(ID),
	foreign key(AID) references ALBUMOK(AID)
);

create table K_NY(
	BID int,
	NyID int,
	darab int,
	primary key(BID,NyID),
	foreign key(BID) references KONYVEK(BID),
	foreign key(NyID) references NYELVEK(NyID)
);

create table TAGDIJ(
	ID int primary key,
	_2020 int,
	_2021 int,
	_2022 int,
	_2023 int,
	foreign key(ID) references TAGOK(ID)
);

create table EMAIL(
	ID int primary key,
	email_1 varchar1 NOT NULL,
	email_2 varchar2,

	foreign key(ID) references TAGOK(ID)
);

create table TELEFON(
	ID int primary key,
	tel_1 varchar1 NOT NULL,
	tel_2 varchar1,
	foreign key(ID) references TAGOK(ID)
);
